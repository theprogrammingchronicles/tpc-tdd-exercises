22  package com.programmingchronicles.tdd.addressbook;
23  
24  import com.programmingchronicles.tdd.domain.Contact;
25  import java.util.ArrayList;
26  import java.util.HashMap;
27  import java.util.List;
28  import java.util.Map;
29  
30  /**
31   * Implementación de un servicio Agenda de Contactos Global
32   * basado en el almacenamiento en memoria.
33   *
34   * <p>
35   * <b>Nueva funcionalidad en un método ya existente:</b><br/>
36   *    Se modifica el método {@link #getContact(java.lang.String)} según
37   *    los requisitos de los nuevos tests.</p>
38   *
39   * <p>
40   * <b>IMPORTANTE:</b><br/>
41   *     Ahora es más importante que nunca hacer fallar los tests antes de
42   *     realizar la implementación, ya que se está ampliando la funcionalidad
43   *     de métodos ya existentes.
44   * </p>
45   *
46   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
47   */
48  public class GlobalAddressBook {
49  
50      private Map<String, Contact> addressBookMap = new HashMap();
51  
52      // PREGUNTA:
53      //   Nos podríamos plantear que la generación de ids ni siquiera es
54      //   responsabilidad del AddressBook, o que es una funcionalidad tán
55      //   generica que si se extrae se puede reutilizar.
56      //
57      // TODO: En un posterior ejemplo aplicaremos esta politica para
58      //       empezar con el tema de los stubs/mocks.
59      private static int nextId = 0;
60  
61      /**
62       * REFACTORIZACION.
63       * <br/>
64       *
65       * <pre>
66       * ¿Es realmente responsabilidad del addContact generar los ids?
67       *  <i>Pregunta</i>: ¿Y si cambio la forma de generar ids?
68       *
69       *  Decidimos que si queremos cambiar el sistema de generación de ids
70       *  no queremos cambiar el sistema de añadir contactos.
71       * </pre>
72       *
73       * @param contact Datos del contacto a añadir
74       * @return Devuelve el id asignado al contacto
75       */
76      public String addContact(Contact contact) {
77          String id = createNewId();
78          
79          contact.setId(id);                
80          addressBookMap.put(id, contact);
81          return id;
82      }
83  
84      /**
85       * Obtiene todos los contactos del sistema.
86       *
87       * @return
88       */
89      public List<Contact> getAll() {
90          List<Contact> contacts = new ArrayList(addressBookMap.size());
91          contacts.addAll(addressBookMap.values());
92          return contacts;
93      }
94  
95      /**
96       * Obtiene el contacto asociado al id entregado. El método se ha modificado
97       * para lanzar una excepción cuando el id es inválido, según los requisitos
98       * impuestos por los nuevos tests.
99       *
100      * @param contactId id del contacto
101      * @return Devuelve el contacto
102      * @throws InvalidIdException Excepción cuando el id es inválido.
103      */
104     public Contact getContact(String contactId) {
105         Contact result = addressBookMap.get(contactId);
106         if(result == null) {
107            throw new InvalidIdException();
108         } 
109         return result;
110     }
111 
112     // Método privado con la única responsabilidad de generar ids
113     private String createNewId() {
114         return Integer.toString(nextId++);
115     }
116 }
