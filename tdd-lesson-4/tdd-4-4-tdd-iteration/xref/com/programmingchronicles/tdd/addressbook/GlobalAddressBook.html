22  package com.programmingchronicles.tdd.addressbook;
23  
24  import com.programmingchronicles.tdd.domain.Contact;
25  import java.util.ArrayList;
26  import java.util.HashMap;
27  import java.util.List;
28  import java.util.Map;
29  
30  /**
31   * Implementación de un servicio Agenda de Contactos Global
32   * basado en el almacenamiento en memoria.
33   *
34   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
35   */
36  public class GlobalAddressBook {
37  
38      private Map<String, Contact> addressBookMap = new HashMap();
39  
40      // Acceso al servicio de generación de ids, que se configurará
41      // mediante Direct Injection (IoC).
42      private IdGenerator idGenerator;
43  
44      /**
45       * Añade un nuevo contacto devolviendo el id generado. El nombre
46       * del contacto es obligatorio, si no se entrega se lanza una excepción.
47       *
48       * @param contact Datos del contacto a añadir
49       * @return Devuelve el id asignado al contacto
50       * @throws InvalidContactException
51       */
52      public String addContact(Contact contact) {
53          // Nombre obligatorio teniendo en cuenta espacios antes y despues.
54          if(contact.getFirstName() == null || contact.getFirstName().trim().length() < 1) {
55             throw new InvalidContactException();
56          }       
57  
58          // Para comprobar duplicados de contactos por nombre, decidimos que
59          // quitamos directamente los espacios para siguientes operaciones.
60          contact.setFirstName(contact.getFirstName().trim());
61          if(checkDuplicate(contact)) {
62             throw new InvalidContactException();
63          }
64  
65          String id = idGenerator.newId();
66  
67          contact.setId(id);
68          addressBookMap.put(id, contact);
69          return id;
70      }
71      
72      /**
73       * Obtiene todos los contactos del sistema.
74       *
75       * @return
76       */
77      public List<Contact> getAll() {
78          List<Contact> contacts = new ArrayList(addressBookMap.size());
79          contacts.addAll(addressBookMap.values());
80          return contacts;
81      }
82  
83      /**
84       * Obtiene el contacto asociado al id entregado.
85       *
86       * @param contactId id del contacto
87       * @return Devuelve el contacto
88       */
89      public Contact getContact(String contactId) {
90          Contact result = addressBookMap.get(contactId);
91          if(result == null) {
92             throw new InvalidIdException();
93          }
94          return result;
95      }        
96  
97      // Método privado obtenido por refactorización. Este método será
98      // testeado indirectamente.
99      private boolean checkDuplicate(Contact checkedContact) {
100         String checkedFirstName = checkedContact.getFirstName();
101         String checkedSurname = checkedContact.getSurname();
102 
103         for(Contact contact: addressBookMap.values()) {
104             String firstName = contact.getFirstName();
105             String surname = contact.getSurname();
106 
107             // El contacto esta repetido si conincide en nombre y apellido, al
108             // añadir el apellido en la comprobación, saltan otros tests y vemos
109             // un caso de tests que no teniamos contemplado.
110             if(checkedFirstName.equalsIgnoreCase(firstName) && checkedSurname.equals(surname)) {
111                return true;
112             }
113         }
114         return false;
115     }
116 
117     /**
118      * Obtiene el generador de ids configurado.
119      *
120      * @return
121      */
122     public IdGenerator getIdGenerator() {
123         return idGenerator;
124     }
125 
126     /**
127      * Configura el generador de ids utilizado.
128      *
129      * @param idGenerator
130      */
131     public void setIdGenerator(IdGenerator idGenerator) {
132         this.idGenerator = idGenerator;
133     }
134 }
