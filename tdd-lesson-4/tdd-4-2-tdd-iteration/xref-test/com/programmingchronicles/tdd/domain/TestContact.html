96      @Test

99          Contact contact = new Contact();
100         contact.setId("id");
101         contact.setFirstName("firstName");
102         contact.setSurname("surname");
103         contact.setPhone("phone");
104         Date expectedDate = new Date();
105         contact.setBirthday(expectedDate);
106 
107 
108         // Aunque no conviente tener mas de un assert en cada test, en este
109         // caso es necesario.
110 
111         // Es la solución al fallo no detectado en "testId"
112 
113         // No tenemos acceso a los atributos, por ser private, y no queremos
114         // romper la encapsulación que proporciona OOP.
115 
116         // Se establecen todas las propiedades en la misma instancia y se prueban
117         // que tienen el valor correcto usando los getters.
118         assertEquals("id", contact.getId());
119         assertEquals("firstName", contact.getFirstName());
