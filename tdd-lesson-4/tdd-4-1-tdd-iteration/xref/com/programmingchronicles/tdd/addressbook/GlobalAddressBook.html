22  package com.programmingchronicles.tdd.addressbook;
23  
24  import com.programmingchronicles.tdd.domain.Contact;
25  import java.util.ArrayList;
26  import java.util.HashMap;
27  import java.util.List;
28  import java.util.Map;
29  
30  /**
31   * Implementación de un servicio Agenda de Contactos Global
32   * basado en el almacenamiento en memoria.
33   *
34   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
35   */
36  public class GlobalAddressBook {
37  
38      private Map<String, Contact> addressBookMap = new HashMap();
39  
40      // Acceso al servicio de generación de ids, que se configurará
41      // mediante Direct Injection (IoC).
42      private IdGenerator idGenerator;
43  
44      /**
45       * Añade un nuevo contacto devolviendo el id generado.
46       *
47       * @param contact Datos del contacto a añadir
48       * @return Devuelve el id asignado al contacto
49       */
50      public String addContact(Contact contact) {
51          String id = idGenerator.newId();
52  
53          contact.setId(id);
54          addressBookMap.put(id, contact);
55          return id;
56      }
57  
58      /**
59       * Obtiene todos los contactos del sistema.
60       *
61       * @return
62       */
63      public List<Contact> getAll() {
64          List<Contact> contacts = new ArrayList(addressBookMap.size());
65          contacts.addAll(addressBookMap.values());
66          return contacts;
67      }
68  
69      /**
70       * Obtiene el contacto asociado al id entregado.
71       *
72       * @param contactId id del contacto
73       * @return Devuelve el contacto
74       */
75      public Contact getContact(String contactId) {
76          Contact result = addressBookMap.get(contactId);
77          if(result == null) {
78             throw new InvalidIdException();
79          }
80          return result;
81      }   
82  
83      /**
84       * Obtiene el generador de ids configurado.
85       *
86       * @return
87       */
88      public IdGenerator getIdGenerator() {
89          return idGenerator;
90      }
91  
92      /**
93       * Configura el generador de ids utilizado.
94       *
95       * @param idGenerator
96       */
97      public void setIdGenerator(IdGenerator idGenerator) {
98          this.idGenerator = idGenerator;
99      }
100 }
