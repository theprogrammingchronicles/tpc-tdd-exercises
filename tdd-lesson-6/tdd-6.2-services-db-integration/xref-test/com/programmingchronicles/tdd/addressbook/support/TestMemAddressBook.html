236     @Test
237     public void testAddDuplicateNameWithNullSurname() {
238         Contact c1 = new Contact();
239         c1.setFirstName("Pedro");
240         c1.setSurname("Ballesteros");
241 
242         Contact c2 = new Contact();
243         c2.setFirstName("Pedro");
244         c2.setSurname(null);
245 
246         String actualId1 = addressBook.addContact(c1);
247         String actualId2 = addressBook.addContact(c2);
248 
249         // Se comprueba que se han añadido los dos.
250         // NOTA: Esto ya lanza un error, el mock que tenemos para crear ids no
251         //       nos sirve, hay que cambiarlo ya que siempre genera el mismo.
252         // Ahora el mock se inicializa así:
253         //       when(generatorMock.newId()).thenReturn("0", "1", "2", "3", "4");
254         List<Contact> contacts = addressBook.getAll();
255         assertEquals(2, contacts.size());
256 
257         // Se verifican los contactos recibidos
258           //  assertEquals("Pedro", contacts.get(1).getFirstName());
259           //  assertEquals("Ballesteros", contacts.get(1).getSurname());
260           //  assertEquals("Pedro", contacts.get(0).getFirstName());
261           //  assertNull(contacts.get(0).getSurname());
262 
263         // ANTIPATRON:
264         //  Con los asserts anteriores el test se hace fragil, se está fijando
265         //  un orden en la respuesta.
266         //  Ningun requisito indica que getAll debe devolver algún orden
267         //  determinado. Y deberiamos hacer el test independiente de que
268         //  como se implemente la construcción de la lista de respuesta.
269 
270         // En este caso particular se está implementando con un HashMap, por lo
271         // que el orden incluso será muy indeterminado.
272 
273         // En este caso el assertEquals(2, contacts.size()) podría ser más que
274         // suficiente ya el método addContact ya ha probado que añade bien
275         // todos los campos en tests anteriores.
276 
277         // Pero si nos queremos asegurarnos hay dos opciones:
278         //
279         //   - Implementar el assert usando el método getContact para
280         //     buscar el contacto que queremos verificar.
281         //
282         //   - O implementar una busqueda del contacto dentro del
283         //     List recibido.
284 
285         // Usamos la primera opción, el getContact ya está testeado.
286         Contact expectedC1 = addressBook.getContact(actualId1);
287         assertEquals("Pedro", expectedC1.getFirstName());
288         assertEquals("Ballesteros", expectedC1.getSurname());
289 
290         Contact expectedC2 = addressBook.getContact(actualId2);
291         assertEquals("Pedro", expectedC2.getFirstName());
292         assertNull(expectedC2.getSurname());
