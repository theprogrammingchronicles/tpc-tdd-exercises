35  /**
36   * Controlador para mostrar y procesar el formulario de añadir contactos.
37   *
38   * <p>
39   * Permitir la configuración de las vistas que utilizará cada controlador.</p>
40   *
41   * <p>
42   * <b>DISEÑO:</b><br/>
43   * Se usa un patrón clásico de procesamiento de formularios
44   * <ul>
45   *    <li>El HTTP GET devuelve el formulario.
46   *    <li>El HTTP POST procesa el formulario.
47   *       <ul>
48   *          <li> Si OK devuelve un redirect al navegador, evitando el doble POST.
49   *          <li> Si Error devuelve de nuevo el formulario indicando los campos erroneos.
50   *       </ul>
51   * </ul>
52   *
53   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
54   */
55  public class AddContactController extends HttpServlet {
56  
57      // Despues de la codificación de ShowContactController, nos damos cuenta
58      // de que es mejor diseño dejar las vistas configurables.
59      // Luego refactorizaremos ShowContactsContrller con la misma idea.
60      private String successedSubmitRedirect;
61      private String formView;
62  
63      // Acceso al servicio de GlobalAddressBook, se configurará
64      // mediante Direct Injection (IoC).
65      private GlobalAddressBook addressBook;
66     
67      /** 
68       * Handles the HTTP <code>GET</code> method.
69       *
70       * @param request servlet request
71       * @param response servlet response
72       * @throws ServletException if a servlet-specific error occurs
73       * @throws IOException if an I/O error occurs
74       */
75      @Override
76      protected void doGet(HttpServletRequest request, HttpServletResponse response)
77      throws ServletException, IOException {
78          getServletContext().getRequestDispatcher(formView).forward(request, response);
79      } 
80  
81      /** 
82       * Handles the HTTP <code>POST</code> method.
83       *
84       * @param request servlet request
85       * @param response servlet response
86       * @throws ServletException if a servlet-specific error occurs
87       * @throws IOException if an I/O error occurs
88       */
89      @Override
90      protected void doPost(HttpServletRequest request, HttpServletResponse response)
91      throws ServletException, IOException {
92         String firstName = request.getParameter("firstName");
93         String surname = request.getParameter("surname");
94         String birthday = request.getParameter("birthday");
95         String phone = request.getParameter("phone");
96  
97         DateFormat dateFormat = new SimpleDateFormat("dd/MM/yyyy");
98  
99         // Se convierten los parametros al domain object que necesita
100        // el servicio address book.
101        Contact contact = new Contact();
102        contact.setFirstName(firstName);
103        contact.setSurname(surname);
104        try {
105            // Ya vemos que esto fallará si no se entrega el birthday, pero
106            // no lo implementamos todavía, ya que en principio no tenemos
107            // un test de prueba de parametros erroneos.
108            contact.setBirthday(dateFormat.parse(birthday));
109        } catch(ParseException ex) {
110             throw new ServletException("Wrong Birthday", ex);
111        }
112        contact.setPhone(phone);
113 
114        // Se inserta el contacto en la agenda
115        addressBook.addContact(contact);
116 
117        // Envia una respuesta de redirección al navegador (evitando el doble post).
118        response.sendRedirect(successedSubmitRedirect);
119     }
120
