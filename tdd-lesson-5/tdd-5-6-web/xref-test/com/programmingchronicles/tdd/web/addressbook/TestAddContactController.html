172     /**
173      * Test de HttpPost: Verifica que el contacto se añade a la agenda
174      * usando el servicio de address book.
175      *
176      * <p>Uso de <b>argThat</b> en los parámetros del <b>verify</b> de
177      * mockito para usar un Hamcrest Custom Matcher.
178      */
179     @Test
180     public void testDoPostAddContact() throws ServletException, IOException, ParseException {
181        // Se construyen mocks de los parametros del "doGet", que es el
182        // método que se quiere testear.
183        HttpServletRequest mockRequest = mock(HttpServletRequest.class);
184        HttpServletResponse mockResponse = mock(HttpServletResponse.class);
185 
186        // Datos del contacto de ejemplo que se añadirá en el POST.
187        Contact expectedContact = new Contact();
188        String fomattedBirthday = "8/1/1974";
189        expectedContact.setFirstName("Pedro");
190        expectedContact.setSurname("Ballesteros");
191        Date expectedBirthday = dateFormat.parse(fomattedBirthday);
192        expectedContact.setBirthday(expectedBirthday);
193        expectedContact.setPhone("610101010");
194 
195        // Se codifican respuestas del "getParameter" con los datos del
196        // contacto de ejemplo a añadir.
197        when(mockRequest.getParameter("firstName")).thenReturn(expectedContact.getFirstName());
198        when(mockRequest.getParameter("surname")).thenReturn(expectedContact.getSurname());
199        when(mockRequest.getParameter("birthday")).thenReturn(fomattedBirthday);
200        when(mockRequest.getParameter("phone")).thenReturn(expectedContact.getPhone());
201 
202        // Ejecución del test.
203        controller.doPost(mockRequest, mockResponse);
204 
205        // Ya no es necesario configurar un ArgumentCaptor.
206        //
207        // Se puede realizar una verificación por interación usando el
208        // IsContactEqual Hamcrest Custom Matcher. El verify comprueba que
209        // el método se ha invocado con el argumento esperado.
210        verify(mockAddressbook).addContact(argThat(contactNoIdEq(expectedContact)));
211 
212        // NOTA:
213        //  Las comprobaciones se simplificacan enormemente, sobre todo cuando
214        //  se deben comprobar tipos de datos muy extensos ya que las operaciones
215        //  de matching se centralizan sin necesidad de tocar el código funcional
216        //  implementando un "equals" adaptado a los tests.
217 
218        /*
219         Versión anterior basasa en ArgumentCaptor:
220 
221            ArgumentCaptor<Contact> argument = ArgumentCaptor.forClass(Contact.class);
222            verify(mockAddressbook).addContact(argument.capture());
223 
224            Contact contact = argument.getValue();
225            assertEquals("Pedro", contact.getFirstName());
226            assertEquals("Ballesteros", contact.getSurname());
227            assertEquals(dateFormat.parse("8/1/1974"), contact.getBirthday());
228            assertEquals("69696969", contact.getPhone());
229        */
230 
231        // Verifica que no se ha intentado crear el modelo de errores
232        verify(mockRequest, never()).setAttribute(eq("errors"), any());
233     }

271     /**
272      * Test Http Post: Implementa un test con todos los parametros vacíos.
273      *
274      * <p>
275      * <b>Refactorizacion:</b><br/>
276      *   Mientras implementamos la funcionalidad que hace pasar a este test,
277      *   nos damos cuenta que se puede aplicar el principio SRP:<br/>
278      *   La responsabilidad de validar y convertir los parametros se puede
279      *   extraer del controlador.</p>
280      *
281      * <p>
282      * Dejamos pendiente este test y su funcionalidad y nos ponemos
283      * con la refactorización, que implica nuevas clases y por tanto
284      * los tests de dichas clases.</p>
285      *
286      * <p>
287      * <b>NOTA:</b> Este es el estado del test despues de varias iteraciones, en
288      *              este caso no se han ido añadiendo mas tests, si no que se ha
289      *              ido actualizando este.</p>
290      *
291      * @throws ServletException
292      * @throws IOException
293      */
294     @Test
295     public void testDoPostNullParams() throws ServletException, IOException {
296        // Mock del Response
297        HttpServletResponse mockResponse = mock(HttpServletResponse.class);
298 
299        // Mock del Request que devuelve null para todos los parametros.
300        HttpServletRequest mockRequest = mock(HttpServletRequest.class);
301        when(mockRequest.getParameter(anyString())).thenReturn(null);
302 
303        // Ejecución del test.
304        controller.doPost(mockRequest, mockResponse);
305 
306        // Verifica que no se ha intentado añadir el contacto ya que
307        // el nombre es obligatorio.
308        verify(mockAddressbook, never()).addContact(any(Contact.class));
309 
310        // Verifica que se ha creado un modelo con los errores con un
311        // setAtribute. Decidimos que se introducirán en un Map<String, String>.
312        ArgumentCaptor<Map> argument = ArgumentCaptor.forClass(Map.class);
313        verify(mockRequest).setAttribute(eq("errors"), argument.capture());
314 
315        // En el verify capturamos los parametros del setAttribute que el
316        // controlador usa sobre el response, para comprobar que se
317        // construye bien el modelo de los errores.
318        Map<String,String> errors = argument.getValue();
319 
320        // REFACTORIZACION:
321        //   Despues de la refactorizacion SRP que hemos realizado creando
322        //   el ContactConvertConverter (que tiene sus propios tests), ya no
323        //   sería tan importante comprobar aqui los errores, con verificar
324        //   que el modelo se introduce en el request podría ser suficiente.
325        assertTrue(errors.containsKey("firstName"));
326        assertFalse(errors.containsKey("surname"));
327        assertFalse(errors.containsKey("birthday"));
328        assertFalse(errors.containsKey("phone"));
329 
330        // IMPORTANTE:
331        //   No se han comprobado los textos de error, en un test unitario no se
332        //   deben probar la IU. Dichas pruebas estarían en tests de IU (test de sistema).
333 
334        // En caso de error la petición se debe dirigir a la vista del formulario
335        // de nuevo, que ahora mostrará los campos de entrda erroneos.
336        verify(mockDispather, times(1)).forward(mockRequest, mockResponse);
337     }
