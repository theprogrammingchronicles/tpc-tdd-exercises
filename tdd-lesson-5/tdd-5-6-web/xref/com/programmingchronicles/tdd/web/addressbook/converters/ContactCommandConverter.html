32  /**
33   * Extrae los parametros de un contacto desde un HTTP Request realizando
34   * la conversión y la validación.

36   * <p>
37   * <b>NOTA:</b> Este código representa el estado de la clase despues de varias

40   * <p>
41   * Esta clase también eliminará espacios en los parametros y detectará
42   * cadenas vacías.</p>

73  public class ContactCommandConverter {
74       // Las fechas se deben entregar en este formato
75       private static DateFormat dateFormat = new SimpleDateFormat("dd/MM/yyyy");
76  
77       private Contact contact = new Contact();
78       private Map<String, String> errors = new HashMap();
79  
80      /**
81       * Realiza la conversión de los parametros de un request en
82       * un Contact, actualizando los errores y el contact del converter.
83       *
84       * <p>
85       * Si un campo es erroneo el resto se siguen comprobando.</p>
86       *
87       * @param request
88       * @param errors
89       * @return true si la conversión no ha dado errores.
90       */
91      public boolean verifyAndConvert(HttpServletRequest request) {
92          boolean result = true;        
93  
94          result &= convertFirstName(request);
95          result &= convertBirthday(request);
96          result &= convertPhone(request);
97          result &= convertSurname(request);
98          return result;        
99      }
100     
101     /**
102      * Devuelve el Contact relleno con los campos que se han podido convertir.
103      *
104      * @return
105      */
106     public Contact getCommand() {
107         return contact;
108     }
109 
110     /**
111      * Devuelve un Map con el nombre de las propiedades de un contact
112      * mapeadas en textos de error.
113      *
114      */
115     public Map<String, String> getErrors() {
116         return errors;
117     }

119     private boolean convertFirstName(HttpServletRequest request) {
120         String firstName = request.getParameter("firstName");
121         
122         // Se eliminan espacios en blanco, considerandolo null si 
123         // al final es una cadena vacía.
124         if(firstName != null) {
125            firstName = firstName.trim();
126            if(firstName.length() == 0) {
127               firstName = null;
128            }
129         }
130 
131         // Comprueba que es un nombre válido ya que es obligatorio.
132         if(firstName == null) {
133            // Habría que tener una internacionalización de textos
134            errors.put("firstName", "El nombre es obligatorio.");
135            return false;
136         }
137 
138         contact.setFirstName(firstName);
139         return true;
140     }
141

147         if(surname != null) {
148            surname = surname.trim();

150               surname = null;
151            }
152         }
153         contact.setSurname(surname);
154         return true;
155     }

162         if(phone != null) {

172     private boolean convertBirthday(HttpServletRequest request) {
173         String birthday = request.getParameter("birthday");
174 
175         // Se eliminan espacios en blanco, considerandolo null si
176         // al final es una cadena vacía.
177         if(birthday != null) {
178            birthday = birthday.trim();
179            if(birthday.length() == 0) {
180               birthday = null;
181            }
182         }
183 
184         // Si el contacto no es null, ni blancos, ni cadena vacía, se
185         // intenta la conversión.
186         if(birthday != null) {
187             try {
188                contact.setBirthday(dateFormat.parse(birthday));
189             } catch(ParseException ex) {
190                errors.put("birthday", "El formato de la fecha debe ser dd/mm/yyyy");
191                return false;
192             }
193         }
194         return true;
195     }
