81      /**
82       * Verifica que funciona correctamente con un contacto completo.
83       *
84       * @throws ParseException
85       */
86      @Test
87      public void testConvertSuccessed() throws ParseException {
88          HttpServletRequest request = mock(HttpServletRequest.class);
89          // Mock del request que devuelve todos los campos del formulario.
90          when(request.getParameter("firstName")).thenReturn("Pedro");
91          when(request.getParameter("surname")).thenReturn("Ballesteros");
92          when(request.getParameter("birthday")).thenReturn("8/1/1974");
93          when(request.getParameter("phone")).thenReturn("69696969");
94  
95          boolean result = converter.verifyAndConvert(request);
96          // Verifica que no falla.
97          assertTrue(result);
98  
99          // Ningún error
100         assertEquals(converter.getErrors(), Collections.emptyMap());
101 
102         // Se obtiene el contacto completo
103         Contact contact = converter.getCommand();
104         assertEquals("Pedro", contact.getFirstName());
105         assertEquals("Ballesteros", contact.getSurname());
106         assertEquals(dateFormat.parse("8/1/1974"), contact.getBirthday());
107         assertEquals("69696969", contact.getPhone());
108     }
