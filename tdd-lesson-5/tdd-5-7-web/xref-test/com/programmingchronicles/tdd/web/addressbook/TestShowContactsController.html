37  /**
38   * Tests del controlador de obtención de contactos.
39   *
40   * <p>
41   * Esta implementacion usa los nuevos Fakes de HttpServletRequest y
42   * HttpServletResponse que a su vez permiten verificar por estado
43   * las redirecciones y los forwards.</p>
44   *
45   * <p>
46   * Ahora el código de los tests disminuye y se hace más claro, al poder
47   * realizar validación por estado.</p>
48   *
49   * <p>
50   * <b>NOTAS RESTRICCIONES:</b><br/>
51   *   En este ejemplo ya empezamos a usar el request para obtener
52   *   el dispatcher. Que era una restriccion que se había impuesto
53   *   solo para mostrar ejemplos de Spy del Object Under Test.</p>
54   *
55   * <p>
56   * <b>DISEÑO:</b> Arquitectura MVC usando tan sólo tecnologías del J2EE:<br/>
57   *     <li>- Modelo: JavaBeans (Java-Old Plain Objects).
58   *     <li>- Controlador: Un Servlet para cada petición web.
59   *     <li>- Vista: JSP (sin nada de lógica, solo JSTL + EL).
60   * </p>
61   *
62   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
63   */
64  public class TestShowContactsController {
65  
66      // Mocks
67      private GlobalAddressBook mockAddressbook;
68      private List<Contact> contactList;
69  
70      // Object Under Test
71      private ShowContactsController controller;
72  
73      @Before
74      public void setUp() {
75          // Ya no construimos el Spy del Controlador, en este ejercicio se ha
76          // eliminado la restricción que se había impuesto, y podemos
77          // usar el request para obtener el requestDispatcher.
78          controller = new ShowContactsController();
79  
80          // Fixtures: Contactos de ejemplo.
81          Contact c1 = new Contact();
82          c1.setId("Id1");
83          Contact c2 = new Contact();
84          c2.setId("Id2");
85  
86          // Lista lista de contactos de ejemplo que devuelve la agenda.
87          contactList = Arrays.asList(c1, c2);
88  
89          // Mock de GlobalAddressBook que devuelve los contactos de ejemplo.
90          mockAddressbook = mock(GlobalAddressBook.class);
91          when(mockAddressbook.getAll()).thenReturn(contactList);
92  
93          // Se configura por Direct Injection el addressbook usado
94          // por el controlador.
95          controller.setAddressBook(mockAddressbook);
96  
97          // La vista ahora es configurable
98          controller.setViewPath("viewPath");
99      }

101     /**
102      * Test del método doGet, que procesa la petición.
103      *
104      * <p>
105      * Al ser una arquitectura MVC, lo unico que hay que testear es:<br/>
106      * <ol>
107      *    <li> El controlador recibe los parametros.
108      *    <li> Construye el modelo de forma adecuada.
109      *    <li> Se redirige la petición a la vista adecuada.
110      * </ol>
111      * </p>
112      * <p>
113      * El controlador se implementará como un Servlet, y la logica se
114      * procesará en un HTTP GET.</p>
115      */
116     @Test
117     public void testDoGet() throws ServletException, IOException {
118 
119        // Fakes de los parametros del doGet y doPost que amplian la interfaz
120        // estandar permitiendo verificaciones sencillas por estado.
121        FakeHttpServletRequest fakeRequest = new FakeHttpServletRequest();
122        FakeHttpServletResponse fakeResponse = new FakeHttpServletResponse();
123 
124        // Ejecución del test.
125        controller.doGet(fakeRequest, fakeResponse);
126 
127        // Diseño de nuestra arquitectura MVC:
128        //   El servlet (controlador) construye el modelo y lo hace
129        //   accesible a la vista, que será una JSP.
130        //
131        //   El modelo se crea como atributos del request (request.setAttribute).
132        //
133        //   En este caso aprobechamos los domain objects del sistema. Otros
134        //   controladores podrían definir modelos propios. Las necesidades
135        //   particulares de la logica de presentación no deben influir en la
136        //   lógica de negocio. El modelo es lógica de presentación, no de
137        //   negocio.
138 
139        // El request y response son los parametros de entrada y salida
140        // del método testado, con los fakes ahora podemos hacer verificación
141        // por estado en lugar de interacción.
142 
143        // Verifica que el modelo se a creado correctamente. Ahora se puede
144        // verificar por estado, porque sabemos que nuestro fake devuelve
145        // valores reales establecidos con el setAttribute.
146        assertSame(fakeRequest.getAttribute("contacts"), contactList);
147 
148        // Verifica que el forward se realiza a la vista correcta, usando
149        // las facilidades del Fake que permiten obtener el path del forward,
150        // en lugar de realizar una validación por interacción.
151        assertEquals("viewPath", fakeResponse.getForwardLocation());
152     }
