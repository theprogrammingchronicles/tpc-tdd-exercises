22  package com.programmingchronicles.addressbook.converters;
23  
24  import com.programmingchronicles.tdd.domain.Contact;
25  import java.text.DateFormat;
26  import java.text.ParseException;
27  import java.text.SimpleDateFormat;
28  import java.util.HashMap;
29  import java.util.Map;
30  import javax.servlet.http.HttpServletRequest;
31  
32  /**
33   * Extrae los parametros de un contacto desde un HTTP Request realizando
34   * la conversión y la validación.
35   *
36   * Esta clase también eliminará espacios en los parametros y detectará
37   * cadenas vacías.
38   *
39   *  - Metodos privados: Se testean de forma indirecta.
40   *
41   * CREACION DE FRAMEWORKS:
42   *   En futuras refactorizaciones podríamos diseñar un mecanismo más
43   *   generico para operaciones de conversión y verificación de parametros,
44   *   por ejemplo usando reflexion para usar propiedades del javabean.
45   *
46   *   Está comenzando a aparecer un framework en nuestra arquitectura:
47   *   En TDD nunca nos ponemos por adelantado a diseñar un framework
48   *   generico que se pueda reutilizar en un futuro.
49   *
50   *   "Se programa para el ahora y nunca para el futuro"
51   *
52   *   Pero en las clases Converter empezamos a vislumbrar que mediante
53   *   la refactorización iterativa estamos dando los primeros pasos
54   *   hacia nuestro propio framework, sin haberlo diseñado por adelantado.
55   *
56   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
57   */
58  public class ContactCommandConverter {
59       // Las fechas se deben entregar en este formato
60       private static DateFormat dateFormat = new SimpleDateFormat("dd/MM/yyyy");
61  
62       // Se centralizan los textos de error que habría que extraer a un
63       // sistema de internacionalización.
64       private static String NOTFOUND_FIRSTNAME = "El nombre es obligatorio.";
65       private static String CONVERT_BIRTHDAY =  "El formato de la fecha debe ser dd/mm/yyyy";
66  
67       private Contact contact = new Contact();
68       private Map<String, String> errors = new HashMap();
69  
70      /**
71       * Realiza la conversión de los parametros de un request en
72       * un Contact, actualizando los errores y el contact del converter.
73       * Si un campo es erroneo el resto se siguen comprobando.
74       *
75       * NOTA: Usando reflection o una utilidad de acceso a propiedades
76       * JavaBean se podría incluso simplificar más.
77       *
78       * @param request
79       * @param errors
80       * @return true si la conversión no ha dado errores.
81       */
82      public boolean verifyAndConvert(HttpServletRequest request) {
83  
84          convertFirstName(request);
85          convertBirthday(request);
86          contact.setSurname(convertParam(request, "surname"));
87          contact.setPhone(convertParam(request, "phone"));
88  
89          // Si la colección de errores está vacía se devuelve true.
90          return errors.isEmpty();
91      }
92  
93      // Devuelve el parametro entregado eliminado los espacios en blanco
94      // y devolviendo null si es cadena vacía o solo blancos.
95      private String convertParam(HttpServletRequest request, String paramName) {
96          String value = request.getParameter(paramName);
97  
98          // Se eliminan espacios en blanco y devuelve null si el parametro
99          // está en blanco.
100         if(value != null) {
101            value = value.trim();
102            if(value.length() == 0) {
103               value = null;
104            }
105         }
106         return value;
107     }
108 
109     // Realiza la conversión y actualización del nombre, actualizando
110     // los posibles errores.
111     private void convertFirstName(HttpServletRequest request) {
112         String firstName = convertParam(request, "firstName");
113         contact.setFirstName(firstName);
114 
115         // El nombre es obligatorio.
116         if(firstName == null) {
117            errors.put("firstName", NOTFOUND_FIRSTNAME);
118         }
119     }
120 
121      // Realiza la conversión y actualización del cumpleaños, actualizando
122      // los posibles errores.
123      private void convertBirthday(HttpServletRequest request) {
124         String birthday = convertParam(request, "birthday");
125 
126         // Solo se intenta convertir si no era null o vacío.
127         if(birthday != null) {
128             try {
129                contact.setBirthday(dateFormat.parse(birthday));
130             } catch(ParseException ex) {
131                errors.put("birthday", CONVERT_BIRTHDAY);
132             }
133         }
134     }
135 
136     /**
137      * Devuelve el Contact relleno con los campos que se han podido convertir.
138      *
139      * @return
140      */
141     public Contact getCommand() {
142         return contact;
143     }
144 
145     /**
146      * Devuelve un Map con el nombre de las propiedades de un contact
147      * mapeadas en textos de error.
148      *
149      */
150     public Map<String, String> getErrors() {
151         return errors;
152     }
153 }
