35  /**
36   * Implementación de AddressBookDao basada en JDBC.
37   *
38   * <p><b>Refactorización</b><br/>
39   * Se realiza una refactorización para eliminar todo el código repetitivo
40   * de manipulación de objetos de acceso a base de datos y control del
41   * cierre de recursos (se disminuyen unas 50 líneas de código).</p>
42   *
43   * <p>El datasource ahora se utiliza a través del template implementado, pero
44   * el DAO se sigue usando exactamente igual a través de la propiedad
45   * <i>datasource</i> proporcionada por la abstracción {@link #getDataSource() }
46   * & {@link #setDataSource(javax.sql.DataSource) }.
47   *
48   * <p>El código repetido se extrae en una clase reutilizable que utiliza el
49   * patrón de diseño <b>Template</b>. También se centraliza la conversión
50   * de claves generadas por la base de datos en claves del objeto del dominio.</p>
51   *
52   * <p>Los test de los DAOs ahora se pueden despreocupar completamente de la
53   * gestión de recursos, que ahora está enteramente probada por test unitarios,
54   * verdaderamente aislados de la base de datos.<p>
55   *
56   * <p><b>Nota:</b><br/>
57   * Existen librerías que ya proporcionan este tipo de templates para JDBC,
58   * como <i>JDBC Spring Templates</i>.</p>
59   *
60   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
61   */
62  public class JdbcAddressBookDao implements AddressBookDao {

64      // El datasource ahora se utiliza a través del template implementado,
65      // pero el DAO se sigue usando exactamente igual a través de la propiedad
66      // datasource (getDataSource & setDataSource).
67      private JdbcDaoTemplate template;
68  
69      // Sentencias SQL utilizadas por el DAO.
70      private static final String SQL_INSERT_CONTACT =
71              "INSERT INTO CONTACTS(FIRSTNAME, SURNAME, BIRTHDAY, PHONE) "
72              + "VALUES(?, ?, ?, ?)";
73      private static final String SQL_SELECT_ALL =
74              "SELECT ID, FIRSTNAME, SURNAME, BIRTHDAY, PHONE FROM CONTACTS";
75      private static final String SQL_SELECT_CONTACT =
76              "SELECT ID, FIRSTNAME, SURNAME, BIRTHDAY, PHONE "
77              + "FROM CONTACTS "
78              + "WHERE ID = ?";
79      private static final String SQL_DELETE_CONTACT =
80              "DELETE FROM CONTACTS "
81              + "WHERE ID = ?";
82  
83      /**
84       * Obtiene el DataSource configurado que se utilizará para la obtención
85       * de conexiones a la base de datos.
86       *
87       * @return
88       */
89      public DataSource getDataSource() {
90          return template.getDatasource();
91      }

93      /**
94       * Configura el DataSource utilizado para la obtención de conexiones a
95       * la base de datos.
96       *
97       * @param ds
98       */
99      public void setDataSource(DataSource ds) {
100         this.template = new JdbcDaoTemplate(ds);
101     }

103     @Override
104     public String addContact(final Contact contact) {
105         List<Number> keys;
106         keys = template.insert(SQL_INSERT_CONTACT,
107             new ParameterMapper() {
108                 @Override
109                 public void mapParameters(PreparedStatement statement) throws SQLException {
110                     statement.setString(1, contact.getFirstName());
111                     statement.setString(2, contact.getSurname());
112                     if (contact.getBirthday() != null) {
113                         statement.setDate(3, new java.sql.Date(contact.getBirthday().getTime()));
114                     } else {
115                         statement.setDate(3, null);
116                     }
117                     statement.setString(4, contact.getPhone());
118                 }
119             });
120         // Como sólo se inserta una fila, sólo debe haber una clave.
121         return keyToId(keys.get(0).longValue());
122     }
123 
124     @Override
125     public List<Contact> getAll() {
126         List<Contact> result;
127         result = template.query(SQL_SELECT_ALL,
128             new ColumnMapper<Contact>() {
129                 @Override
130                 public Contact mapColumns(ResultSet rsContacts) throws SQLException {
131                     Contact contact = new Contact();
132 
133                     contact.setId(keyToId(rsContacts.getLong("ID")));
134                     contact.setFirstName(rsContacts.getString("FIRSTNAME"));
135                     contact.setSurname(rsContacts.getString("SURNAME"));
136                     contact.setBirthday(rsContacts.getDate("BIRTHDAY"));
137                     contact.setPhone(rsContacts.getString("PHONE"));
138 
139                     return contact;
140                 }
141             });
142         return result;
143     }

145     @Override
146     public Contact getContact(String contactId) {
147 
148         // Si el id es invalido el DAO se comporta igual que si no existiera
149         // el contacto, devolviendo null.
150         final long key;
151         try {
152             key = idToKey(contactId);
153         } catch(NumberFormatException ex) {
154             return null;
155         }
156 
157         List<Contact> contacts;
158         contacts = template.query(SQL_SELECT_CONTACT,
159             new ParameterMapper() {
160                 @Override
161                 public void mapParameters(PreparedStatement statement) throws SQLException {
162                     statement.setLong(1, key);
163                 }
164             },
165             new ColumnMapper<Contact>() {
166                 @Override
167                 public Contact mapColumns(ResultSet rsContacts) throws SQLException {
168                     Contact contact = new Contact();
169 
170                     contact.setId(keyToId(rsContacts.getLong("ID")));
171                     contact.setFirstName(rsContacts.getString("FIRSTNAME"));
172                     contact.setSurname(rsContacts.getString("SURNAME"));
173                     contact.setBirthday(rsContacts.getDate("BIRTHDAY"));
174                     contact.setPhone(rsContacts.getString("PHONE"));
175 
176                     return contact;
177                 }
178             });
179 
180         // Si el contacto no existe se de debe devolver null
181         if (contacts.isEmpty()) {
182             return null;
183         } else {
184             // Se devuelve solo el primer contacto ya que la SQL solo
185             // busca un contacto por id.
186             return contacts.get(0);
187         }
188     }
189 
190     @Override
191     public void deleteContact(String contactId) {
192         // Si el id es invalido el DAO se comporta igual que si no existiera.
193         final long key;
194         try {
195             key = idToKey(contactId);
196         } catch(NumberFormatException ex) {
197             return;
198         }
199 
200         template.update(SQL_DELETE_CONTACT,
201             new ParameterMapper() {
202                 @Override
203                 public void mapParameters(PreparedStatement statement) throws SQLException {
204                     statement.setLong(1, key);
205                 }
206             });
207     }
208 
209     private String keyToId(long key) {
210         return Long.toString(key);
211     }
212 
213     private long idToKey(String id) throws NumberFormatException {
214         return Long.parseLong(id);
215     }
