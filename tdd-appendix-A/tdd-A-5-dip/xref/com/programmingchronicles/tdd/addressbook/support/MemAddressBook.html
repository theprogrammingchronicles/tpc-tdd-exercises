33  /**
34   * Implementación de un servicio Agenda de Contactos Global basado en el
35   * almacenamiento en memoria.
36   *
37   * Desacoplando la instanciación mediante Direct Injection
38   *
39   * La responsabilidad de crear y configurar las dependencias se eleva a
40   * un nivel superior. El cliente de GlobalAddressBook decide que dependencias
41   * utilizará la instancia que necesita, es decir, configura parte de su
42   * comportamiento en función de las caracteristicas deseadas.
43   *
44   * En este entorno la clase MemGlobalAddressBook o FakeGlobalAddressBook son
45   * completamente independientes de cualquier implementación concreta de
46   * IdGenerator, solo saben que se usará un IdGenerator, cualquier implementación
47   * existente le servirá, y con cualquiera funcionará correctamente, es el
48   * cliente el que decide que le conviene.
49   *
50   * MUY IMPORTANTE
51   * La configuración de dependencias es un detalle de implementación, MemGlobalAddressBook
52   * necesita un IdGenerator, pero DBGlobalAddressBook no lo necesita. Nunca debemos
53   * meter métodos de configuración en la interfaz GlobalAddressBook, sería meter
54   * detalles de implementación. Solo metemos lo generico y lo que es parte real
55   * de la API y no de la implementación.
56   *
57   * TESTEABLE
58   * Ahora MemAddressBook está mejor preparada para el unit testing, podemos
59   * aislar los tests de las dependencias, no los basamos ni en RandomIdGenerator
60   * ni en IncrementIdGenerator, solo queremos probar que el código de esta clase
61   * es válido sin importarnos el de otras clases.
62   *
63   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
64   */
65  public class MemAddressBook implements GlobalAddressBook {
66  
67      private Map<String, Contact> addressBookMap = new HashMap();
68  
69      // Acceso un servicio de generación de ids
70      private IdGenerator idGenerator;
71  
72      // Constructor por defecto sin IdGenerator, que se puede configurar
73      // modificando accediendo a setIdGenerator.
74      public MemAddressBook() {      
75      }
76  
77      // Direct Injection basado en constructor, se debe entregar
78      // una implementación de la interfaz IdGenerator.
79      public MemAddressBook(IdGenerator idGenerator) {
80          this.idGenerator = idGenerator;
81      }
82  
83      // Direct Injection basado en propiedad. Permite modificar
84      // el IdGenerator usado por MemAddressBook
85      public void setIdGenerator(IdGenerator idGenerator) {
86          this.idGenerator = idGenerator;
87      }
88  
89      // Acceso al IdGenerator configurado actualmente
90      public IdGenerator getIdGenerator() {
91          return idGenerator;
92      }
93     
94      /**
95       * Añade un nuevo contacto devolviendo el id generado.
96       *
97       * @param contact Datos del contacto a añadir
98       * @return Devuelve el id asignado al contacto
99       */
100     @Override
101     public String addContact(Contact contact) {
102         String id = idGenerator.newId();
103         
104         // Como el servicio de generación de ids puede proporcionar
105         // ids repetidos, se debe comprobar si el id ya fue asignado.
106 
107         // Se podría implementar una lógica de reintentos, pero finalmente
108         // decidimos que es mejor informar del error y que sean los
109         // clientes los encargados de decidir la politica de reintentos.
110         if(addressBookMap.get(id) != null) {
111             throw new InvalidIdException();
112         }
113 
114         contact.setId(id);                
115         addressBookMap.put(id, contact);
116         return id;
117     }
118 
119     /**
120      * Obtiene el contacto asociado al id entregado.
121      *
122      * @param contactId id del contacto
123      * @return Devuelve el contacto
124      */
125     @Override
126     public Contact getContact(String contactId) {
127         Contact result = addressBookMap.get(contactId);
128         if(result == null) {
129            throw new InvalidIdException();
130         } 
131         return result;
132     }
133 
134     /**
135      * Obtiene todos los contactos del sistema.
136      *
137      * @return
138      */
139     @Override
140     public List<Contact> getAll() {
141         // TESTS: En este proyecto no tiene tests, pero es que ha sido copiado
142         //        de una versión más avanzada de los ejercicios de TDD.
143         List<Contact> contacts = new ArrayList(addressBookMap.size());
144         contacts.addAll(addressBookMap.values());
145         return contacts;
146     }
