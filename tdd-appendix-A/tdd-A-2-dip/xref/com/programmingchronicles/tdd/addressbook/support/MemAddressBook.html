22  package com.programmingchronicles.tdd.addressbook.support;
23  
24  import com.programmingchronicles.tdd.addressbook.GlobalAddressBook;
25  import com.programmingchronicles.tdd.addressbook.IdGenerator;
26  import com.programmingchronicles.tdd.addressbook.InvalidIdException;
27  import com.programmingchronicles.tdd.domain.Contact;
28  import java.util.HashMap;
29  import java.util.Map;
30  
31  /**
32   * Implementación de un servicio Agenda de Contactos Global basado en el
33   * almacenamiento en memoria.
34   *
35   * Desacoplando las dependencias entre detalles de implementación usando
36   * las interfaces como abstracciones intermedias.
37   *
38   * Ahora MemGlobalAddressBook ya no depende de una implementación concreta
39   * de IdGenerator, si no de una interfaz que nunca cambiar. Aunque la
40   * implementación entregada contenga otros métodos publicos o interfaces
41   * sabemos que esta clase unicamente utilizará la API IdGenerator.
42   *
43   * Es decir, ya no depende de detalles de implementación, si no de una
44   * abstracción intermedia.
45   *
46   * Podemos cambiar una implementación por otra, sabiendo que no será
47   * necesario cambiar ni una sola linea de código de acceso a IdGenerator.
48   *
49   * AUN SE PUEDE DESACOPLAR MAS
50   *
51   * Por muchos interfaces intermedios que se tengan, MemAddressBook necesita
52   * acceso a una implementación concreta para crear la instancia.
53   *
54   *  <code>idGenerator = new RandomIdGenerator();</code>
55   *
56   * Aunque el cambio ahora tendrá menos impactos, necesitamos seguir
57   * cambiando el código de MemAddressBook si queremos cambiar su implementación.
58   *
59   * ¿Hemos ganado mucho usando interfaces?
60   *
61   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
62   */
63  public class MemAddressBook implements GlobalAddressBook {
64  
65      private Map<String, Contact> addressBookMap = new HashMap();
66  
67      // Acceso un servicio de generación de ids 
68      private IdGenerator idGenerator;
69  
70      public MemAddressBook() {
71          idGenerator = new RandomIdGenerator();
72      }
73     
74      /**
75       * Añade un nuevo contacto devolviendo el id generado.
76       *
77       * @param contact Datos del contacto a añadir
78       * @return Devuelve el id asignado al contacto
79       */
80      @Override
81      public String addContact(Contact contact) {
82          String id = idGenerator.newId();
83          
84          // Como el servicio de generación de ids puede proporcionar
85          // ids repetidos, se debe comprobar si el id ya fue asignado.
86  
87          // Se podría implementar una lógica de reintentos, pero finalmente
88          // decidimos que es mejor informar del error y que sean los
89          // clientes los encargados de decidir la politica de reintentos.
90          if(addressBookMap.get(id) != null) {
91              throw new InvalidIdException();
92          }
93  
94          contact.setId(id);                
95          addressBookMap.put(id, contact);
96          return id;
97      }
98  
99      /**
100      * Obtiene el contacto asociado al id entregado.
101      *
102      * @param contactId id del contacto
103      * @return Devuelve el contacto
104      */
105     @Override
106     public Contact getContact(String contactId) {
107         Contact result = addressBookMap.get(contactId);
108         if(result == null) {
109            throw new InvalidIdException();
110         } 
111         return result;
112     }   
113 }
