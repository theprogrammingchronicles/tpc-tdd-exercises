22  package com.programmingchronicles.tdd.horoscope;
23  
24  import java.util.Calendar;
25  import java.util.Date;
26  import java.util.Locale;
27  
28  /**
29   * Implementación del horoscopo que mide la suerte en función del
30   * día de nacimiento.
31   * <p>
32   * <b>NOTA</b>: Se ha implementado de forma un tanto extraña para explicar
33   *              algunos conceptos de unit testing.
34   * </p>
35   *
36   * <u>IMPLEMENTACION</u>:
37   * <pre>
38   * - Las fechas se pueden introducir en tres formatos y el
39   *   y se detecta automaticamente el formato:
40   *     + Corto: "12/10/2010"
41   *     + Largo: "12 de octubre de 2010"
42   *     + Día: "10" (sería el día 10 del mes y año actual)
43   *
44   * - Si el cumpleaños es en verano (Junio, Julio y Agosto):
45   *    + Si es domingo la suerte es 10.
46   *    + Si no es domingo la suerte es 8.
47   *
48   * - Si el cumpleaños es en invierno (Diciembre, Enero, Febrero):
49   *    + Si es navidad (del 25 al 6) la suerte es 9.
50   *    + Si no es navidad la suerte es 6.
51   *
52   * - El resto del año la suerte es 1.
53   * </pre>
54   * 
55   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
56   */
57  public class Horoscope {
58      private static Calendar calendar = Calendar.getInstance(new Locale("es"));
59  
60      SmartDateParser parser = new SmartDateParser();
61      int luck;
62  
63      /**
64       * Realiza las predicciones según la fecha de nacimiento.
65       *
66       * @return fecha de nacimiento en los formatos soportados
67       */
68      public void makePretictions(String birthday) {
69          Date date = parser.parse(birthday);
70  
71          calendar.setTime(date);
72  
73          // La clase calendar numera los meses de 0 a 11.
74          makePredictions(calendar.get(Calendar.DATE), calendar.get(Calendar.MONTH) + 1,
75                          calendar.get(Calendar.DAY_OF_WEEK));
76      }
77  
78      /**
79       * Acceso a la suerte que predice nuestro horoscopo de 1 a 10.
80       *
81       * @return suerte
82       */
83      public int getLuck() {
84          return luck;
85      }
86  
87      // Método auxiliar que realiza las predicciones con las fechas
88      // ya parseadas y descompuestas.
89      //
90      // Como siempre los MÉTODOS PRIVADOS se prueban indirectamente a través
91      // de la API pública de la clase.
92      private void makePredictions(int day, int month, int weekDay) {
93  
94          if(month >= 6 && month <= 8) {
95             // Verano
96             if(weekDay == Calendar.SUNDAY) {
97                 luck = 10;
98             } else {
99                 luck = 8;
100            }
101 
102         } else if(month == 12 || month <= 2) {
103            // Invierto
104            if((month==12 && day >= 25) || isJanuaryChristmas(day, month)) {
105               // Navidad
106               luck = 9;
107            } else {
108               luck = 6;
109            }
110         } else {
111            luck =1;
112         }
113     }
114 
115     // Metodo auxiliar que decide si es un día de navidad
116     // del mes de febrero (no tengo ni idea de porque se
117     // ha extraido esta condición de los ifs :)
118     private boolean isJanuaryChristmas(int day, int month) {
119         return month==1 && day <= 6;
120     }
121 }
