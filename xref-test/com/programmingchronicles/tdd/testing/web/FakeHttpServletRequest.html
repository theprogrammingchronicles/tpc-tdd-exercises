38  /**
39   * Fake de HttpServletRequest que añade metodos para crear parametros y que
40   * no forman parte de la interfaz. Nos permiten evitar la verificación
41   * por interación.
42   *
43   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
44   */
45  public class FakeHttpServletRequest implements HttpServletRequest {
46  
47      private Map<String, String> params = new HashMap();
48      private Map<String, Object> attributes = new HashMap();
49  
50      /**
51       * Devuelve los parametros establecidos con setParameter
52       *
53       * @param name
54       * @return
55       */
56      @Override
57      public String getParameter(String name) {
58          return params.get(name);
59      }
60  
61      /**
62       * Métodos de ayuda del fake que no forman parte de la interfaz
63       * de la que se está creando el fake.
64       *
65       * @param name
66       */
67      public void setParameter(String name, String value) {
68          params.put(name, value);
69      }
70  
71      @Override
72      public Object getAttribute(String name) {
73         return attributes.get(name);
74      }
75  
76      @Override
77      public void setAttribute(String name, Object o) {
78         attributes.put(name, o);
79      }

81      /**
82       * Devuelve un fake de RequestDispatcher que permite verificar
83       * el forward. El request dispatcher entregado solo actualiza
84       * el response cuando realmente se hace el forward.
85       *
86       * @param path
87       * @return
88       */

90      public RequestDispatcher getRequestDispatcher(String path) {
91          return new FakeRequestDispatcher(path);
92      }
