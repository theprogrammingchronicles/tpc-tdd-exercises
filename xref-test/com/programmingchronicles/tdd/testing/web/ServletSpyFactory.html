22  package com.programmingchronicles.tdd.testing.web;
23  
24  import javax.servlet.ServletContext;
25  import javax.servlet.http.HttpServlet;
26  import static org.mockito.Mockito.*;
27  
28  /**
29   * Factory para la creación de spies del controlador entregado, clase
30   * de utilidades para el teting.
31   *
32   * <p>
33   * <b>REFACTORIZACION:</b></p>
34   *
35   * <p>
36   * Esta clase permite agrupar y reutilizar la funcionalidad que vemos
37   * repetida en los tests de los controladores.</p>
38   *
39   * <p>
40   * En este caso es una utilidad que permite crear un spy del controlador 
41   * entregando (un servlet) creando mocks para el ServletContext</p>
42   *
43   * <p>
44   * Esta clase se podrían extraer en un jar o librería de utilidades que
45   * se podrian añadir con scope test en los proyectos que quieran realizar
46   * este tipo de tests en los servlets.</p>
47   *
48   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
49   */
50  public class ServletSpyFactory<T extends HttpServlet> {
51  
52       private T mockController;
53       private ServletContext mockContext;
54  
55      /**
56       * Crea un un spy o mock parcial del servlet entregado codificando
57       * un stub del método getServletContext para que devuelva un un mock
58       * que a su vez devuelve un mock de requestDispatcher.
59       *
60       * <p>
61       * Los mocks que devuelven mocks no son muy recomendables, suelen ser
62       * indicar fallos de diseño, o como en este caso que estamos usamos un
63       * framework de terceros que no está preparado para el testing
64       * (ver documentación de <a href="http://www.mockito.com">mockito</a>).</p>
65       *
66       * @param <T> Tipo del controlador. Debe heredar de HttpServlet.
67       * @param controller Instancia del controlador sobre el que se crea el Spy.
68       */
69      public ServletSpyFactory(T controller) {
70          // Mocks del context del servlet.
71          mockContext = mock(ServletContext.class);
72  
73          // Crea un spy del controlador para que devuelva el mock del context
74          // en el método getServletContext.
75          mockController = spy(controller);
76          doReturn(mockContext).when(mockController).getServletContext();
77      }
78  
79      /**
80       * Devuelve el Spy creado para el controlador entregado.
81       *
82       */
83      public T getMockController() {
84          return mockController;
85      }
86  
87      /**
88       * Devuelve el Mock creado para el ServletContext.
89       *
90       */
91      public ServletContext getMockContext() {
92          return mockContext;
93      }    
94  }
