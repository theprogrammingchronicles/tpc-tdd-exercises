22  package com.programmingchronicles.tdd.horoscope;

24  import com.programmingchronicles.tdd.horoscope.DateParserException;
25  import com.programmingchronicles.tdd.horoscope.SmartDateParser;
26  import java.util.Calendar;
27  import java.util.Date;
28  import java.util.Locale;
29  import org.junit.Before;
30  import org.junit.Test;
31  import static org.junit.Assert.*;
32  
33  /**
34   * Test para la clase encargada de parsear un texto en varios
35   * formatos diferentes.
36   *
37   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
38   */
39  public class TestSmartDateParser {
40  
41      private SmartDateParser dateParser = new SmartDateParser();
42      private Calendar calendar = Calendar.getInstance(new Locale("es"));
43  
44      @Before
45      public void setUp() {
46          calendar.clear();
47      }
48  
49      @Test
50      public void testParseShortFormat() {
51          String dateTest = "2/4/2010";
52  
53          calendar.set(2010, 4 - 1, 2);
54          Date expected = calendar.getTime();
55  
56          assertEquals(expected, dateParser.parse(dateTest));
57      }
58  
59      @Test
60      public void testParseFormatWithZeros() {
61          String dateTest = "02/04/2010";
62  
63          calendar.set(2010, 4 - 1, 2);
64          Date expected = calendar.getTime();
65  
66          assertEquals(expected, dateParser.parse(dateTest));
67      }
68  
69      @Test
70      public void testParseLongFormat() {
71          String dateTest = "2 de abril de 2010";
72  
73          calendar.set(2010, 4 - 1, 2);
74          Date expected = calendar.getTime();
75  
76          assertEquals(expected, dateParser.parse(dateTest));
77      }
78  
79      @Test
80      public void testParseMonthDayFormat() {
81          String day = "10";
82  
83          calendar.setTime(new Date());
84          calendar.clear(Calendar.HOUR);
85          calendar.clear(Calendar.MINUTE);
86          calendar.clear(Calendar.SECOND);
87          calendar.clear(Calendar.MILLISECOND);
88  
89          calendar.set(Calendar.DATE, 10);
90          Date expected = calendar.getTime();
91  
92          assertEquals(expected, dateParser.parse(day));
93      }
94  
95      @Test(expected = DateParserException.class)
96      public void testParseInvalidDate() {
97          dateParser.parse("invaliddate");
98      }
99  }
