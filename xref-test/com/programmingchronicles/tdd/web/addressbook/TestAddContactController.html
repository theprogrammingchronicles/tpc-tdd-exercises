89      @Test
90      public void testInit() throws ServletException {
91  
92          ServletConfig config = mock(ServletConfig.class);
93          // Se codifica una respuesta para los parametros de configuración del
94          // servlet. Se configuran las vistas en el web.xml y el init debe
95          // inicializar su configuración. No hay posibilidad de IoC.
96          when(config.getInitParameter("formView")).thenReturn("formViewValue");
97          when(config.getInitParameter("successedSubmitRedirect")).thenReturn("successedSubmitRedirectValue");
98  
99          // El mock del contexto devuelve una instancia de GlobalAddressBook
100         // que a su vez es un mock. Solo hay que verificar que el init es
101         // capaz de obtener esta instancia correctamente. No se usa una
102         // instancia concreta, el servlet no está ligado a implementaciones.
103         ServletContext context = mock(ServletContext.class);
104         when(config.getServletContext()).thenReturn(context);        
105         when(context.getAttribute("globalAddressBook")).thenReturn(mockAddressbook);
106 
107         // Se instancia un controlador no inicializado.
108         AddContactController notInitController = new AddContactController();
109         notInitController.init(config);
110 
111         // Se verifica la configuración
112 
113         // Se comprueba que el addressbook se ha configurado con lo que devuelve
114         // el contexto. Se comprueba que es instancia de GlobalAddressBook, no
115         // se liga a implementaciones específicas.
116         assertTrue("init addressbook failed", notInitController.getAddressBook() instanceof GlobalAddressBook);
117 
118         // Verifica que las vistas se configuran con el valor devuelto por
119         // getInitParameter del servlet. Los valores reales de las vistas
120         // no se deben testear ya que son configuraciones de despliegue.
121         assertEquals("formViewValue", notInitController.getFormView());
122         assertEquals("successedSubmitRedirectValue", notInitController.getSuccessedSubmitRedirect());
123     }

152        // Fakes de los parametros del doGet y doPost que amplian la interfaz
153        // estandar permitiendo verificaciones sencillas por estado.
154        FakeHttpServletRequest fakeRequest = new FakeHttpServletRequest();
155        FakeHttpServletResponse fakeResponse = new FakeHttpServletResponse();
156 
157        // Ahora los fackes nos permiten configurar los parametros de entrada
158        // del método testeado. Esto no lo permitían los mocks del contenedor

169        ArgumentCaptor<Contact> argument = ArgumentCaptor.forClass(Contact.class);
170        verify(mockAddressbook).addContact(argument.capture());

172        // En el verify anterior capturamos el argumento con el que se llama

176        assertEquals("Ballesteros", contact.getSurname());
177        assertEquals(dateFormat.parse("8/1/1974"), contact.getBirthday());
178        assertEquals("69696969", contact.getPhone());
179 
180        // Verifica con el fake que no se ha intentado crear el modelo de errores

188        assertEquals("redirectPath", fakeResponse.getSendRedirectLocation());

191     // Test Http Post: Verifica un post con todos los parametros vacíos.
192     //
193     @Test
194     public void testDoPostNullParams() throws ServletException, IOException {
195        // Fakes de los parametros del doGet y doPost que amplian la interfaz
196        // estandar permitiendo verificaciones sencillas por estado.
197        FakeHttpServletRequest fakeRequest = new FakeHttpServletRequest();
198        FakeHttpServletResponse fakeResponse = new FakeHttpServletResponse();     
199 
200        // Ejecución del test para comprobar que se establece el modelo
201        // en este caso los errores.
202        controller.doPost(fakeRequest, fakeResponse);
203 
204        // Verifica que no se ha intentado añadir el contacto ya que
205        // el nombre es obligatorio.
206        verify(mockAddressbook, never()).addContact(any(Contact.class));
207 
208        // Verifica que se ha creado un modelo con los errores con un
209        // setAtribute. Decidimos que se introducirán en un Map<String, String>.
210 
211        // Ahora se puede verificar por estado, porque sabemos que nuestro fake
212        // devuelve valores reales establecidos con el setAttribute.
213        Map<String, String> errors = (Map<String, String>)fakeRequest.getAttribute("errors");
214        assertTrue(errors.containsKey("firstName"));
215        assertFalse(errors.containsKey("surname"));
216        assertFalse(errors.containsKey("birthday"));
217        assertFalse(errors.containsKey("phone"));
218 
219        // IMPORTANTE:
220        //   No se han comprobado los textos de error, en un test unitario no se
221        //   deben probar la IU. Dichas pruebas estarían en tests de IU (test de sistema).
222 
223        // En caso de error la petición se debe dirigir a la vista del formulario
224        // de nuevo, que ahora mostrará los campos de entrada erroneos.
225 
226        // Ahora se puede verificar por estado ya que el fake del response nos
227        // permite acceder al path con el que se hizo el forward.
228        assertEquals("formViewPath", fakeResponse.getForwardLocation());
229     }
