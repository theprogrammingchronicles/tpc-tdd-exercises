39  /**
40   * Tests del controlador de obtención de contactos.
41   *
42   * Se implementa el test del init para validar que el servlet
43   * inicializa bien su configuración.
44   *
45   * FUTURA REFACTORIZACION:
46   *    Implementar una arquitectura MVC de Controlador Frontal de forma
47   *    que los controladores no tengan que ser servlets.
48   *
49   * DISEÑO:
50   *    Arquitectura MVC usando sólo tecnologías nativas del J2EE:
51   *       - Modelo: JavaBeans (Java-Old Plain Objects).
52   *       - Controlador: Servlets.
53   *       - Vista: JSP (sin lógica: solo JSTL + EL).
54   *
55   * @author Pedro Ballesteros <pedro@theprogrammingchronicles.com>
56   */
57  public class TestShowContactsController {
58  
59      // Mocks
60      private GlobalAddressBook mockAddressbook;
61      private List<Contact> contactList;
62  
63      // Object Under Test
64      private ShowContactsController controller;
65  
66      @Before
67      public void setUp() {      
68          controller = new ShowContactsController();
69  
70          // Fixtures: Contactos de ejemplo.
71          Contact c1 = new Contact();
72          c1.setId("Id1");
73          Contact c2 = new Contact();
74          c2.setId("Id2");
75  
76          // Lista lista de contactos de ejemplo que devuelve la agenda.
77          contactList = Arrays.asList(c1, c2);
78  
79          // Mock de GlobalAddressBook que devuelve los contactos de ejemplo.
80          mockAddressbook = mock(GlobalAddressBook.class);
81          when(mockAddressbook.getAll()).thenReturn(contactList);
82  
83          // Se configura por Direct Injection el addressbook usado
84          // por el controlador.
85          controller.setAddressBook(mockAddressbook);
86  
87          // La vista ahora es configurable
88          controller.setViewPath("viewPath");
89      }
90  
91      // El controlador necesita buscar los servicios en el init, ya que
92      // no se pueden configurar por Direct Injection. Es el contenedor de
93      // servlet el que crea los controladores.
94      @Test
95      public void testInit() throws ServletException {
96          ServletConfig config = mock(ServletConfig.class);
97  
98          // Se codifica una respuesta para los parametros de configuración del
99          // servlet. Se configuran las vistas en el web.xml y el init debe
100         // inicializar su configuración. No hay posibilidad de IoC.
101         when(config.getInitParameter("view")).thenReturn("viewPathValue");
102 
103         // Se configura un mock del contexto que devuelve una instancia
104         // de GlobalAddressBook como un mock. Solo hay que verificar
105         // que el init configura bien el controlador, por tanto no hace
106         // falta una implementación real de GlobalAddressBook.
107         ServletContext context = mock(ServletContext.class);
108         when(config.getServletContext()).thenReturn(context);     
109         when(context.getAttribute("globalAddressBook")).thenReturn(mockAddressbook);
110 
111         // Se instancia un controlador no inicializado.
112         ShowContactsController notInitController = new ShowContactsController();
113         notInitController.init(config);
114 
115         // Se verifica la configuración
116 
117         // Se comprueba que el addressbook se ha configurado con lo que devuelve
118         // el contexto. Se comprueba que es instancia de GlobalAddressBook, no
119         // se liga a implementaciones específicas.
120         assertTrue("init addressbook failed", notInitController.getAddressBook() instanceof GlobalAddressBook);
121 
122         // Verifica que las vistas se configuran con el valor devuelto por
123         // getInitParameter del servlet. Los valores reales de las vistas
124         // no se deben testear ya que son configuraciones de despliegue.
125         assertEquals("viewPathValue", notInitController.getViewPath());
126     }

128     /**
129      * Al ser una arquitectura MVC, lo unico que hay que testear es:
130      *      - El controlador recibe los parametros.
131      *      - Construye el modelo de forma adecuada.
132      *      - Se redirige la petición a la vista adecuada.
133      *
134      * El controlador se implementará como un Servlet, y la logica se
135      * procesará en un HTTP GET.
136      */
137     @Test
138     public void testDoGet() throws ServletException, IOException {
139 
140        // Fakes de los parametros del doGet y doPost que amplian la interfaz
141        // estandar permitiendo verificaciones sencillas por estado.
142        FakeHttpServletRequest fakeRequest = new FakeHttpServletRequest();
143        FakeHttpServletResponse fakeResponse = new FakeHttpServletResponse();
144 
145        // Ejecución del test.
146        controller.doGet(fakeRequest, fakeResponse);
147 
148        // Diseño de nuestra arquitectura MVC:
149        //   El servlet (controlador) construye el modelo y lo hace
150        //   accesible a la vista, que será una JSP.
151        //
152        //   El modelo se crea como atributos del request (request.setAttribute).
153        //
154        //   En este caso aprobechamos los domain objects del sistema. Otros
155        //   controladores podrían definir modelos propios. Las necesidades
156        //   particulares de la logica de presentación no deben influir en la
157        //   lógica de negocio. El modelo es lógica de presentación, no de
158        //   negocio.
159 
160        // El request y response son los parametros de entrada y salida
161        // del método testado, con los fakes ahora podemos hacer verificación
162        // por estado en lugar de interacción.
163 
164        // Verifica que el modelo se a creado correctamente. Ahora se puede
165        // verificar por estado, porque sabemos que nuestro fake devuelve
166        // valores reales establecidos con el setAttribute.
167        assertSame(fakeRequest.getAttribute("contacts"), contactList);
168 
169        // Verifica que el forward se realiza a la vista correcta, usando
170        // las facilidades del Fake que permiten obtener el path del forward,
171        // en lugar de realizar una validación por interacción.
172        assertEquals("viewPath", fakeResponse.getForwardLocation());
173     }
