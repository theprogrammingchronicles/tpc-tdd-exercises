156     /**
157      * Crea un stub del datasource que devuelve siempre un wrapper de la conexión
158      * entregada, que ignora las transacciones y el cierre de conexión, para que
159      * los tests puedan realizar el rollback de las modificaciones.
160      *
161      * @see ConnectionWrapper
162      *
163      * @param connection
164      * @return
165      * @throws SQLException
166      */
167     protected DataSource newSingleNonTransactionalDataSource(Connection connection) throws SQLException {
168 
169         // Crea un wrapper de la conexión para ignorar transacciones.
170         final Connection nonTransactionalConnection = new ConnectionWrapper(connection) {
171             // Ya no es necesario verificar el close del conection, ya que ahora
172             // no es responsabilidad del DAO.
173             @Override
174             public void close() throws SQLException {
175             }
176 
177             @Override
178             public void commit() throws SQLException {
179             }
180 
181             @Override
182             public void rollback() throws SQLException {
183             }
184 
185             @Override
186             public void setAutoCommit(boolean autoCommit) throws SQLException {
187             }
188         };
189 
190         // Crea un stub de un datasource que devuelve el wrapper de la conexión.
191         DataSource mockDatasource = mock(DataSource.class);
192 
193         // El datasource devuelve el connection spy.
194 
195         // Ya no es necesario verificar el close del conection, ya que ahora
196         // no es responsabilidad del DAO.
197         when(mockDatasource.getConnection()).thenReturn(nonTransactionalConnection);
198         when(mockDatasource.getConnection(anyString(), anyString()))
199              .thenReturn(nonTransactionalConnection);
200 
201         return mockDatasource;
202     }

536 /**
537  * Implementación de un custom matcher de contact que permite realizar
538  * comparaciones ignorando o no el id.
539  *
540  * <p>El matcher se puede usar con el <b>assertThat</b> de JUnit o para el
541  * argument matching de los mocks de mockito.</p>
542  *
543  * <p>Aunque se podría hacer lo mismo implementando el método <i>Contact#equals</i>,
544  * de esta forma podemos hacer comparaciones no ligadas a la lógica, como por
545  * ejemplo ignorar el id en las comparaciones.</p>
546  *
547  * <p>Permite reducir el código de los tests, sobre todo cuando hay que
548  * verificar listas y colecciones.</p>
549  *
550  * @throws ParseException
551  * @throws SQLException
552  */
553 class IsContactEqual extends TypeSafeMatcher<Contact> {
554     private final Contact c2;
555     private final boolean ignoreId;
556 
557     public IsContactEqual(Contact contact, boolean ignoreId) {
558         this.c2 = contact;
559         this.ignoreId = ignoreId;
560     }
561 
562     public IsContactEqual(Contact contact) {
563         this.c2 = contact;
564         this.ignoreId = false;
565     }
566 
567     /**
568      * Se sobrescribe para definir cuando dos contactos son el mismo.
569      *
570      * @param c1
571      * @return
572      */
573     @Override
574     public boolean matchesSafely(Contact c1) {
575         boolean result;
576         result = equals(c1.getFirstName(), c2.getFirstName())  &&
577                  equals(c1.getSurname(), c2.getSurname()) &&
578                  equals(c1.getBirthday(), c2.getBirthday()) &&
579                  equals(c1.getPhone(), c2.getPhone());
580 
581         if(!ignoreId) {
582             result = result && equals(c1.getId(), c2.getId());
583         }
584         return result;
585     }
586 
587     /**
588      * Los hamcrest matchers dan información muy precisa de la causa del
589      * fallo. Sobrescribiendo este método se puede mostrar el expected
590      * object de forma detallada.
591      *
592      * <p>
593      * Aunque se debería implementar el toString del Contact para mostrar
594      * información completa.</p>
595      *
596      * @param description
597      */
598     @Override
599     public void describeTo(Description description) {
600         description.appendText("a contact with ")
601                    .appendValueList("[", ", ", "]",
602                           c2.getId(), c2.getFirstName(), c2.getSurname(),
603                           c2.getBirthday(), c2.getPhone());
604     }
605 
606     /**
607      * Matcher que compara dos contactos completos.
608      *
609      * @param contact
610      * @return
611      */
612     @Factory
613     public static Matcher<Contact> contactEq(Contact contact) {
614         return new IsContactEqual(contact);
615     }
616 
617     /**
618      * Matcher que compara dos contactos ignorando los ids.
619      *
620      * @param contact
621      * @return
622      */
623     @Factory
624     public static Matcher<Contact> contactNoIdEq(Contact contact) {
625         return new IsContactEqual(contact, true);
626     }

628     private static boolean equals(Object o1, Object o2) {
629         return o1 == o2 || (o1 != null && o1.equals(o2));
630     }
631 }

633 /**
634  * Crea un conexión implementando el <b>Delegation pattern</b> sobre la
635  * conexión real entregada.
636  *
637  * <p>
638  * El patrón de delegación permitirá implementar un partial mock de la conexión
639  * que ignorará los métodos de gestión de transacciones y el cierre de la
640  * conexión, para que los test puedan realizar el rollback de las
641  * modificaciones de los DAOs.</p>
642  *
643  * <p>
644  * <b>IMPORTANTE:</b><br/>
645  *    No se puede utilizar {@link org.mockito.Mockito#spy(java.lang.Object)} ya
646  *    que se necesita control sobre la conexión real y mockito se basa en la
647  *    creación de proxies que realizan una copia de la instancia. Por tanto
